Not to be messed up with getProfile from weight.js

Params

uuid

@type string
Use :
The persons to get the profiles from.

apiKey

@type : string
Use :
The hypixel api key.

Returns

profile

@type object
Use :
The current profile. 

import { getProfile } from "dist/minecraft/chatHandling/commands/networth.js";
import getUuid from 'dist/minecraft/chatHandling/utils/getUuid.js';
import { API_KEY } from 'dist/minecraft/chatHandling/utils/getUuid.js'

const uuid = getUUid("puppynuff");

const profile = await getProfile(uuid, API_KEY)
